 26 var sys = sys || {};

 28 /** LocalStorage is a local storage component.
 29 */
 30 sys.localStorage = window.localStorage;

 32 
 33 /** Capabilities
 34 */
 35 Object.defineProperties(sys,
 36 {
 37 	"capabilities" : {
 38 		get : function(){
 39 			var capabilities = {"canvas":true};
 40 
 41 			// if (window.DeviceOrientationEvent!==undefined || window.OrientationEvent!==undefined)
 42 			//   capabilities["accelerometer"] = true;
 43 
 44 			if( 'ontouchstart' in document.documentElement )
 45 				capabilities["touches"] = true;
 46 			else if( 'onmouseup' in document.documentElement )
 47 				capabilities["mouse"] = true;
 48 
 49 			if( 'onkeyup' in document.documentElement )
 50 				capabilities["keyboard"] = true;
 51 
 52 			return capabilities;
 53         },
 54 		enumerable : true,
 55 		configurable : true
 56 	},
 57 	"os" : {
 58 		get : function() {
 59 			var iOS = ( navigator.userAgent.match(/(iPad|iPhone|iPod)/i) ? true : false );
 60 			var isAndroid = navigator.userAgent.match(/android/i) || navigator.platform.match(/android/i) ? true : false;
 61 			var OSName=navigator.appVersion;
 62 			if (navigator.appVersion.indexOf("Win")!=-1)
 63 				OSName="Windows";
 64 			else if (navigator.appVersion.indexOf("Mac")!=-1)
 65 				OSName="OS X";
 66 			else if (navigator.appVersion.indexOf("X11")!=-1)
 67 				OSName="UNIX";
 68 			else if (navigator.appVersion.indexOf("Linux")!=-1)
 69 				OSName="Linux";
 70 			else if( iOS )
 71 				OSName = "iOS";
 72 			else if( isAndroid )
 73 				OSName = "Android";
 74 		},
 75 		enumerable : true,
 76 		configurable : true
 77 	},
 78 	"platform" : {
 79 		get : function(){
 80 			return "browser";
 81 		},
 82 		enumerable : true,
 83 		configurable : true
 84 	},
 85 	"version" : {
 86 		get : function(){
 87 			return cc.ENGINE_VERSION;
 88 		},
 89 		enumerable : true,
 90 		configurable : true
 91 	}
 92 });
